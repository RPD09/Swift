	//EXERCICIO 1
	
	# Gerando dados fictícios
	set.seed(123)
	altura <- rnorm(100, mean = 170, sd = 10)
	idade <- rnorm(100, mean = 35, sd = 12)
	erro <- rnorm(100, mean = 0, sd = 5)
	peso <- 50 + 0.8 * altura - 0.2 * idade + erro
	
	# Criando um dataframe
	dados <- data.frame(altura, idade, peso)
	
	# Ajuste do modelo de regressão linear múltipla
	modelo <- lm(peso ~ altura + idade, data = dados)
	summary(modelo)
	
	# Visualização: Altura vs Peso e Idade vs Peso com linha de regressão
	par(mfrow = c(1, 2))  # Dividir em dois gráficos
	plot(dados$altura, dados$peso, main = "Altura vs Peso", xlab = "Altura (cm)", ylab = "Peso (kg)")
	abline(lm(peso ~ altura, data = dados), col = "red")
	
	plot(dados$idade, dados$peso, main = "Idade vs Peso", xlab = "Idade (anos)", ylab = "Peso (kg)")
	abline(lm(peso ~ idade, data = dados), col = "blue")
	
	# Previsão para novos indivíduos
	novos_individuos <- data.frame(altura = c(160, 180, 175), idade = c(40, 25, 30))
	previsoes <- predict(modelo, novos_individuos)
	print(previsoes)
	
	# Cálculo e interpretação dos resíduos
	residuos <- resid(modelo)
	plot(residuos, main = "Resíduos do Modelo", ylab = "Resíduos", xlab = "Índices")
	
	# Verificação das suposições da regressão
	par(mfrow = c(2, 2))
	plot(modelo)

	peso <- 50 + 0.8 * altura - 0.2 * idade + rnorm(100, mean = 0, sd = 10)  # Peso em kg
	
	# Criar um data frame
	dados <- data.frame(altura, idade, peso)

	modelo <- lm(peso ~ altura + idade, data = dados)

	summary(modelo)

	# a) Altura vs Peso com linha de regressão

	novos_individuos <- data.frame(altura = c(160, 180, 175), idade = c(40, 25, 30))

	novos_individuos$peso_previsto <- previsoes

	# 6. Interpretação dos coeficientes do modelo
	coeficientes <- summary(modelo)$coefficients
	print(coeficientes)
	
	
	//Informações do modelo
	3. Análise do Modelo
	Após ajustar o modelo de regressão linear múltipla, o resumo do modelo (obtido com summary(modelo)) fornecerá as seguintes informações:
	
	Coeficientes do Modelo:
	
	Intercepto (β₀): Representa o peso previsto quando a altura e a idade são zero. Pode não ter um significado prático, mas é parte da equação do modelo.
	Altura (β₁): O coeficiente da altura indica quanto o peso muda para cada aumento de uma unidade na altura (em cm). Um coeficiente positivo indica que, em média, à medida que a altura aumenta, o peso também aumenta.
	Idade (β₂): O coeficiente da idade indica quanto o peso muda para cada aumento de uma unidade na idade (em anos). Um coeficiente negativo sugere que, em média, à medida que a idade aumenta, o peso tende a diminuir.
	
	Além dos coeficientes, você vai analisar:

	Valor de R²: Um valor de R² próximo de 1 indica que o modelo é capaz de explicar a maior parte da variação no peso com base na altura e na idade, enquanto valores mais baixos sugerem que o modelo não está capturando todos os fatores que afetam o peso.
	Outros Parâmetros Importantes:
	
	Estatística F: Avalia se o modelo como um todo é estatisticamente significativo.
	P-valores: Usados para verificar a significância estatística dos coeficientes. Um p-valor abaixo de 0.05 geralmente indica que o coeficiente é estatisticamente significativo.

	Coeficiente de Altura: O coeficiente da altura é aproximadamente 0.86. Isso indica que, para cada aumento de 1 cm na altura, o peso previsto aumenta em média 0.86 kg, mantendo a idade constante. Como o p-valor é menor que 0.05, podemos afirmar que a altura tem uma influência significativa sobre o peso.
	
	Coeficiente de Idade: O coeficiente da idade é aproximadamente -0.16. Isso indica que, para cada aumento de 1 ano na idade, o peso previsto diminui em média 0.16 kg, mantendo a altura constante. O p-valor também indica que esta relação é estatisticamente significativa.
